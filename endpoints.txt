# models.py
from django.db import models

class Category(models.Model):
    name = models.CharField(max_length=100)
    image = models.URLField()
    icon = models.CharField(max_length=10)

class Product(models.Model):
    name = models.CharField(max_length=200)
    category = models.ForeignKey(Category, on_delete=models.CASCADE)
    weight = models.CharField(max_length=50)
    price = models.DecimalField(max_digits=10, decimal_places=2)
    original_price = models.DecimalField(max_digits=10, decimal_places=2)
    image = models.URLField()
    badge = models.CharField(max_length=50, blank=True)
    description = models.TextField()
    nutrition_info = models.JSONField()
    is_active = models.BooleanField(default=True)

class Order(models.Model):
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    items = models.JSONField()  # Cart items
    total_amount = models.DecimalField(max_digits=10, decimal_places=2)
    status = models.CharField(max_length=50, default='pending')
    created_at = models.DateTimeField(auto_now_add=True)

# models.py
class BulkOrder(models.Model):
    ORDER_TYPES = [
        ('restaurant', 'Restaurant/Hotel'),
        ('corporate', 'Corporate/Cafeteria'),
        ('retailer', 'Retailer/Reseller'),
        ('event', 'Event/Catering'),
    ]
    
    order_id = models.CharField(max_length=50, unique=True)
    order_type = models.CharField(max_length=20, choices=ORDER_TYPES)
    client_name = models.CharField(max_length=200)
    contact_person = models.CharField(max_length=100)
    mobile = models.CharField(max_length=15)
    gst_number = models.CharField(max_length=20, blank=True)
    address = models.TextField()
    pricing_tier = models.CharField(max_length=20)  # retail/bulk/wholesale
    subtotal = models.DecimalField(max_digits=10, decimal_places=2)
    discount = models.DecimalField(max_digits=10, decimal_places=2)
    tax = models.DecimalField(max_digits=10, decimal_places=2)
    grand_total = models.DecimalField(max_digits=10, decimal_places=2)
    delivery_date = models.DateField()
    delivery_slot = models.CharField(max_length=20)
    status = models.CharField(max_length=20, default='pending')
    staff = models.ForeignKey(User, on_delete=models.CASCADE)
    created_at = models.DateTimeField(auto_now_add=True)

class BulkOrderItem(models.Model):
    bulk_order = models.ForeignKey(BulkOrder, related_name='items', on_delete=models.CASCADE)
    product = models.ForeignKey(Product, on_delete=models.CASCADE)
    quantity = models.PositiveIntegerField()
    unit_price = models.DecimalField(max_digits=10, decimal_places=2)
    total_price = models.DecimalField(max_digits=10, decimal_places=2)

# views.py
@api_view(['POST'])
@permission_classes([IsStaffUser])
def create_bulk_order(request):
    data = request.data
    
    # Generate order ID
    order_id = f"BULK-{timezone.now().strftime('%Y%m%d')}-{BulkOrder.objects.count() + 1:04d}"
    
    # Create order
    order = BulkOrder.objects.create(
        order_id=order_id,
        order_type=data['client_type'],
        client_name=data['client']['name'],
        contact_person=data['client']['contact_person'],
        mobile=data['client']['mobile'],
        gst_number=data['client'].get('gst', ''),
        address=data['client']['address'],
        pricing_tier=data['pricing_tier'],
        subtotal=data['totals']['subtotal'],
        discount=data['totals']['discount'],
        tax=data['totals']['tax'],
        grand_total=data['totals']['grand_total'],
        delivery_date=data['delivery']['date'],
        delivery_slot=data['delivery']['slot'],
        staff=request.user
    )
    
    # Create items
    for item_data in data['items']:
        BulkOrderItem.objects.create(
            bulk_order=order,
            product_id=item_data['id'],
            quantity=item_data['qty'],
            unit_price=item_data['price'],
            total_price=item_data['price'] * item_data['qty']
        )
    
    return Response({
        'success': True,
        'order_id': order_id,
        'message': 'Bulk order created successfully'
    })
